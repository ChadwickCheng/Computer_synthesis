📖 SQL进阶知识点详解

1️⃣～3️⃣ 表连接 (JOIN)

-- 内连接（只返回满足条件的行） SELECT * FROM Teacher t INNER JOIN Stu s ON t.tec_id = s.tec_id; -- 隐式写法 SELECT * FROM Teacher t, Stu s WHERE t.tec_id = s.tec_id; -- 左外连接（保留左表所有行） SELECT * FROM Teacher t LEFT JOIN Stu s ON t.tec_id = s.tec_id; -- 右外连接（保留右表所有行） SELECT * FROM Teacher t RIGHT JOIN Stu s ON t.tec_id = s.tec_id;
💡 记忆口诀：INNER → 交集；LEFT → 左表全保留；RIGHT → 右表全保留；无匹配处填 NULL

4️⃣ 建表与约束条件

CREATE TABLE Teacher ( tec_id INT AUTO_INCREMENT PRIMARY KEY, -- 主键，唯一非空，自增 tname VARCHAR(50) NOT NULL, -- 非空 email VARCHAR(100) UNIQUE, -- 唯一 salary DECIMAL(10,2) CHECK(salary > 0), -- 范围检查 dept VARCHAR(50) DEFAULT '未分配' -- 默认值 ); CREATE TABLE Stu ( stu_id INT AUTO_INCREMENT PRIMARY KEY, sname VARCHAR(50) NOT NULL, tec_id INT, FOREIGN KEY (tec_id) REFERENCES Teacher(tec_id) -- 外键 );
💡 外键让两张表保持关联：学生表的 tec_id 指向教师表，插入时数据库自动校验合法性

5️⃣ 数据库用户与权限管理

-- 创建用户 CREATE USER IF NOT EXISTS 'zhangsan'@'localhost' IDENTIFIED BY 'pwd123'; -- 授权（WITH GRANT OPTION 允许继续授权） GRANT SELECT, INSERT ON school.* TO 'zhangsan'@'localhost'; GRANT ALL PRIVILEGES ON school.* TO 'admin'@'%' WITH GRANT OPTION; -- 回收权限 REVOKE INSERT ON school.* FROM 'zhangsan'@'localhost';
💡 '@localhost' 限制只能本机连接；'@%' 允许任意主机；WITH GRANT OPTION 让被授权者可以继续授权别人

6️⃣ 事务 (Transaction) 与 ACID

START TRANSACTION; INSERT INTO account(id, balance) VALUES(1, 1000); UPDATE account SET balance = balance - 500 WHERE id = 1; COMMIT; -- 成功提交 -- 出错时回滚 START TRANSACTION; UPDATE account SET balance = balance - 500 WHERE id = 1; -- 此处发生错误... ROLLBACK; -- 撤销所有更改
💡 事务就像"打包操作"：要么全成功，要么全失败，保证数据不会只做一半

7️⃣ 索引 (Index)

-- 创建普通 B树索引 CREATE INDEX idx_sname ON Stu(sname); -- 唯一索引 CREATE UNIQUE INDEX idx_email ON Teacher(email); -- 复合索引（多列） CREATE INDEX idx_dept_salary ON Teacher(dept, salary); -- 删除索引 DROP INDEX idx_sname ON Stu;
💡 索引 = 书的目录，查找快但占空间；主键自动建索引；不要过多添加索引，写操作会变慢

8️⃣ 视图 (View)

-- 创建视图：隐藏薪资敏感信息 CREATE VIEW TeacherPublic AS SELECT tec_id, tname, dept FROM Teacher; -- 创建视图：高薪教师 CREATE VIEW HighSalaryTeacher AS SELECT tname, dept, salary FROM Teacher WHERE salary > 8000; -- 像查普通表一样使用视图 SELECT * FROM TeacherPublic; SELECT * FROM HighSalaryTeacher;
💡 视图是虚拟表，不存储数据；对外只暴露必要列，隐藏敏感字段，简化复杂查询

9️⃣ 存储过程 (Stored Procedure)

CREATE OR REPLACE PROCEDURE GetTeacherInfo (p_tno IN INT, p_tname OUT VARCHAR2, p_salary OUT DECIMAL) IS BEGIN SELECT tname, salary INTO p_tname, p_salary FROM Teacher WHERE tec_id = p_tno; EXCEPTION WHEN NO_DATA_FOUND THEN DBMS_OUTPUT.PUT_LINE('未找到教师: ' || p_tno); WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('错误: ' || SQLERRM); END GetTeacherInfo;
💡 参数方向：IN = 输入参数（调用方传入）；OUT = 输出参数（过程写回）；INOUT = 既输入又输出

🔟 触发器 (Trigger)

CREATE OR REPLACE TRIGGER InsertUserLog AFTER INSERT ON users FOR EACH ROW -- 行级触发器 BEGIN INSERT INTO user_logs(user_id, action, log_time) VALUES (:NEW.user_id, 'INSERT', SYSDATE); END;
💡 :NEW 引用新行数据，:OLD 引用旧行数据；BEFORE/AFTER 决定何时触发；FOR EACH ROW 为行级（影响几行触发几次）
📋 触发器执行日志 (user_logs)

1️⃣1️⃣ 作业调度 (Job Scheduling)

-- Oracle DBMS_SCHEDULER 创建作业 BEGIN DBMS_SCHEDULER.CREATE_JOB( job_name => 'DAILY_BACKUP', job_type => 'PLSQL_BLOCK', job_action => 'BEGIN backup_database(); END;', start_date => SYSTIMESTAMP, repeat_interval => 'FREQ=DAILY; BYHOUR=2', -- 每天凌晨2点 enabled => TRUE ); END; -- MySQL EVENT CREATE EVENT daily_cleanup ON SCHEDULE EVERY 1 DAY STARTS '2024-01-01 02:00:00' DO DELETE FROM logs WHERE log_time < DATE_SUB(NOW(), INTERVAL 30 DAY);
💡 作业调度 = 数据库的"定时闹钟"；常用于：每日备份、清理过期日志、定时生成报表
